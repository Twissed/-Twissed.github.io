    """Remove weight of the distribution. Zero method.
    
    Use
    ---
    
    * Define a new weight `weight_norm`.
    * For each particle.
        * if w > weight_norm
            * Create N new particles with same position and momenta.
            with `N = int(np.floor(w/weight_norm))` 

    Args:
        step (Step class): Step class
        weight_norm (float, optional): New weight. Defaults to 1.

    Returns:
        Step class: A new Step class
    """
    
    verbose = kwargs.get('verbose', True)
    
    x_new = []
    y_new = []
    z_new = []
    ux_new = []
    uy_new = []
    uz_new = []
    w_new = []

    for id in range(step.N):
        if step.w[id] > weight_norm:
            w_t = step.w[id]
            N = int(np.floor(w_t/weight_norm)) 
            
            for j in range(N):
                w_new.append(weight_norm)
                x_new.append(step.x[id])
                y_new.append(step.y[id])
                z_new.append(step.z[id])
                ux_new.append(step.ux[id])
                uy_new.append(step.uy[id])

    step_new.getbeam(verbose=verbose)

    """Remove weight of the distribution. Zero method with random for ratio.
    
    Might create more weight than original distribution.
    
    Use
    ---
    
    * Define a new weight `weight_norm`.
    * For each particle.
        * if w > weight_norm
            * Create N new particles with same position and momenta.
            with `N = int(np.floor(w/weight_norm))` 

    Args:
        step (Step class): Step class
        weight_norm (float, optional): New weight. Defaults to 1.

    Returns:
        Step class: A new Step class
    """
    
    verbose = kwargs.get('verbose', True)
    
    x_new = []
    y_new = []
    z_new = []
    ux_new = []
    uy_new = []
    uz_new = []
    w_new = []

    for id in range(step.N):
        w_t = step.w[id]
        N_int = int(np.floor(w_t/weight_norm)) 
        
        for j in range(N_int):
            w_new.append(weight_norm)
            x_new.append(step.x[id])
            y_new.append(step.y[id])
            z_new.append(step.z[id])
            ux_new.append(step.ux[id])
            uy_new.append(step.uy[id])
            uz_new.append(step.uz[id])
            
        N_rat = (w_t/weight_norm) - np.floor(w_t/weight_norm)
        
        if N_rat > random.random():
            w_new.append(weight_norm)
            x_new.append(step.x[id])
            y_new.append(step.y[id])
            z_new.append(step.z[id])
            ux_new.append(step.ux[id])
            uy_new.append(step.uy[id])
            uz_new.append(step.uz[id])
            
    
     
    step_new = Step()
       
    step_new.setpositionsmomenta(x_new, y_new, z_new, ux_new, uy_new, uz_new, np.array(w_new) * np.sum(step.w) / np.sum(np.array(w_new)))
    
    step_new.getbeam(verbose=verbose)  
    
    if verbose == True:                
        print(f"Nparticle obtained: {step_new.N}")
        
        print(f"Emittance x - Old: {step.emit_norm_rms_x}, New: {step_new.emit_norm_rms_x}")
        print(f"Emittance y - Old: {step.emit_norm_rms_y}, New: {step_new.emit_norm_rms_y}")
        print(f"Energy mean - Old: {step.Ek_avg}, New: {step_new.Ek_avg}")

        print(f"Charge - Old: {step.charge}, Saved: {np.sum(np.array(w_new)) * const.e * 1e12 }, New: {step_new.charge}")

    return step_new




[docs]def noweight_matsigma(step, N, **kwargs):
    """Remove weight of the distribution. Method with tranfer matrice.
    
    Use
    ---
    ...

    .. math::

        \sigma = \begin{vmatrix}
        \langle x  \cdot x \rangle & \langle x  \cdot x' \rangle & \langle x  \cdot y \rangle & \langle x  \cdot y' \rangle & \langle x  \cdot z \rangle & \langle x  \cdot z' \rangle  \\
        \langle x' \cdot x \rangle & \langle x' \cdot x' \rangle & \langle x' \cdot y \rangle & \langle x' \cdot y' \rangle & \langle x' \cdot z \rangle & \langle x' \cdot z' \rangle \\
        \langle y  \cdot x \rangle & \langle y  \cdot x' \rangle & \langle y  \cdot y \rangle & \langle y  \cdot y' \rangle & \langle y  \cdot z \rangle & \langle y  \cdot z' \rangle  \\
        \langle y' \cdot x \rangle & \langle y' \cdot x' \rangle & \langle y' \cdot y \rangle & \langle y' \cdot y' \rangle & \langle y' \cdot z \rangle & \langle y' \cdot z' \rangle \\
        \langle z  \cdot x \rangle & \langle z  \cdot x' \rangle & \langle z  \cdot y \rangle & \langle z  \cdot y' \rangle & \langle z  \cdot z \rangle & \langle z  \cdot z' \rangle  \\
        \langle z' \cdot x \rangle & \langle z' \cdot x' \rangle & \langle z' \cdot y \rangle & \langle z' \cdot y' \rangle & \langle z' \cdot z \rangle & \langle z' \cdot z' \rangle 
        \end{vmatrix}


    Args:
        step (Step class): Orignial Step class
        N (int): Approximate number of particle wanted.
    """
    
    
    verbose = kwargs.get('verbose', True)
    
    wtot = np.sum(step.w)
    
    # Normalisation de la distribution 6D
    Part_pt = np.concatenate((np.reshape( (step.x  - step.x_ave )/step.sigma_x ,(1,step.N)),
                          np.reshape( (step.y  - step.y_ave )/step.sigma_y ,(1,step.N)),
                          np.reshape( (step.z  - step.z_ave )/step.sigma_z ,(1,step.N)),
                          np.reshape( (step.ux - step.ux_ave)/step.sigma_ux,(1,step.N)),
                          np.reshape( (step.uy - step.uy_ave)/step.sigma_uy,(1,step.N)),
                          np.reshape( (step.uz - step.uz_ave)/step.sigma_uz,(1,step.N))), axis=0)
    
    # Calcul de la matrice Sigma de la distribution 6D
    sig = np.zeros((6,6))
    for i in range(6):
        for j in range(6):
            sig[i,j] = np.sum( step.w*Part_pt[i,:]*Part_pt[j,:])/wtot
            
    # Recherche de la matrice de transfert qui transforme la matrice sigma en l'identit
    T = np.zeros((6,6))
    for i in range(0,6):
        for j in range(0,i):
            T[i,j] = (sig[j,i]-np.sum(T[i,0:j-1]*T[j,0:j-1]))/T[j,j]
        
        T[i,i] = np.sqrt(sig[i,i]-np.sum(T[i,0:i-1]**2))
     
    # Decouplage de la distribution des particules 
    Part_pt2 = np.zeros_like(Part_pt)
    Part_pt2 = np.matmul(np.linalg.inv(T), Part_pt)
    
    # Tirage de la distribution equi-poids
    pt_norm = np.zeros(step.N)
    pt_norm = step.w / wtot * N # Nombre de particules de poids identiques par particule initiale
    dist = step.N**(-1/6)/3
    Part2 = np.zeros((6,int(np.floor(2.*N)))) # 2 * N particles for buffer
    
    #  Generation de la distribution equi-poids
    j = 0
    for i in range(step.N):
        
        if pt_norm[i] > 1:
            # Si plus d'une particule de m�me poids pour une particule initiale
            
            np_macro = int(np.floor(pt_norm[i]));
            if random.random() <= (pt_norm[i] -np_macro):
                np_macro = np_macro+1
            
            for k in range(np_macro):
                Part2[:,j] = Part_pt2[:,i] + np.random.randn(6,) * dist
                j = j+1
            
        else:
            #  Si plus d'une particule de m�me poids pour une particule initiale
            if random.random() <= pt_norm[i]:  
                Part2[:,j] = Part_pt2[:,i]
                j=j+1

    N_new = j
    
    # New particles
    Part = np.matmul(T,Part2[:,0:N_new])
    
    # Denormalisation
    Part[0,:] = step.sigma_x  * Part[0,:] + step.x_ave
    Part[1,:] = step.sigma_y  * Part[1,:] + step.y_ave
    Part[2,:] = step.sigma_z  * Part[2,:] + step.z_ave
    Part[3,:] = step.sigma_ux * Part[3,:] + step.ux_ave
    Part[4,:] = step.sigma_uy * Part[4,:] + step.uy_ave
    Part[5,:] = step.sigma_uz * Part[5,:] + step.uz_ave
    
         
    step_new = Step()
        
    step_new.setpositionsmomenta(Part[0,:], 
                                Part[1,:],
                                Part[2,:],
                                Part[3,:],
                                Part[4,:],
                                Part[5,:],
                                np.ones(N_new) * wtot / N_new)

    step_new.getbeam(verbose=verbose)  
    
    if verbose == True:                
        print(f"Nparticle obtained: {step_new.N}")
        
        print(f"Emittance x - Old: {step.emit_norm_rms_x}, New: {step_new.emit_norm_rms_x}")
        print(f"Emittance y - Old: {step.emit_norm_rms_y}, New: {step_new.emit_norm_rms_y}")
        print(f"Energy mean - Old: {step.Ek_avg}, New: {step_new.Ek_avg}")

        print(f"Charge - Old: {step.charge}, New: {step_new.charge}")
        
    return step_new
