[docs]def weighted_ave(a: float, weights: float, empty=np.nan) -> float:
    r"""
    Calculate the weighted average of array a.

    .. math::

        \langle {\bf a} \rangle = \frac{\sum_i w_i a_i}{\sum_i w_i} \, ,

    Args:
        arr (float): 1D numpy array
        weights (float): 1D numpy array

    Returns:
        float: weighted average.
    """
    # Check arrays size
    if np.shape(np.asarray(a)) != np.shape(np.asarray(weights)):
        raise TypeError("ERROR: data and weights do not have the same length!")

    # Check if input contains data
    if not np.any(weights) and not np.any(a):
        # If input is empty return empty
        return empty

    else:
        # Calculate the weighted average
        return np.average(a, weights=weights)




[docs]def weighted_std(arr, weights, empty=np.nan, verbose=False):
    """
    Calculate the weighted standard deviation.

    Args:
        arr (float): 1D numpy array
        weights (float): 1D numpy array

    Returns:
        float: weighted average.
    """
    try:
        # Check arrays size
        if np.shape(np.asarray(arr)) != np.shape(np.asarray(weights)):
            raise TypeError("ERROR: data and weights do not have the same length!")

        # Check if input contains data
        if not np.any(weights) and not np.any(arr):
            # If input is empty return empty
            return empty
        else:
            # Calculate the weighted standard deviation
            average = np.average(arr, weights=weights)
            variance = np.average((arr - average) ** 2, weights=weights)
            return np.sqrt(variance)
    except:
        if verbose:
            print("error weighted_std")
        return -1




[docs]def weighted_med(a, weights, verbose=False):
    """
    Compute the weighted median

    Args:
        a (float): 1D numpy array
        weights (float): 1D numpy array

    Returns:
        float: median
    """
    try:
        quantile = 0.5
        if not isinstance(a, np.matrix):
            a = np.asarray(a)
        if not isinstance(weights, np.matrix):
            weights = np.asarray(weights)
        if a.shape != weights.shape:
            raise TypeError("the length of data and weights must be the same")
        ind_sorted = np.argsort(a)
        sorted_data = a[ind_sorted]
        sorted_weights = weights[ind_sorted]

        Sn = np.cumsum(sorted_weights)
        # Center and normalize the cumsum (i.e. divide by the total sum)
        Pn = (Sn - 0.5 * sorted_weights) / Sn[-1]
        # Get the value of the weighted median
        return np.interp(quantile, Pn, sorted_data)

    except:
        if verbose:
            print("error weighted_med")
        return -1




[docs]def weighted_mad(a, w):
    """
    Compute the weighted median absolute

    Args:
        a (float): 1D numpy array
        weights (float): 1D numpy array

    Returns:
        float: median
    """
    med = weighted_med(a, w)
    mad = weighted_med(np.abs(a - med), w)
    return mad



def hist1D_weights(xplotdata, w, xrange=[None, None], bins=100, plot=None, ax=None):
    xname = "x"

    df = pd.DataFrame({xname: xplotdata, "w": w})
    if not xrange[0] == None:
        df = df.drop(df[df[xname] < xrange[0]].index)
    if not xrange[1] == None:
        df = df.drop(df[df[xname] > xrange[1]].index)

    H, xedges = np.histogram(
        df[xname].to_numpy(), bins=bins, weights=df["w"].to_numpy()
    )

    H = H.T * const.e * 1e12

    if not plot == None:
        if ax == None:
            fig, ax = plt.subplots(figsize=(6, 6), dpi=100, tight_layout=True)
        if plot == "plot":
            ax.plot(xedges[:-1], H)
        elif plot == "hist":
            _ = ax.hist(df[xname].to_numpy(), bins=bins, weights=df["w"].to_numpy())

    return H, xedges[:-1]


def hist2D_weights(
    xplotdata,
    yplotdata,
    w,
    xrange=[None, None],
    yrange=[None, None],
    bins=[100, 100],
    ax=None,
    cmap="inferno",
    iscbar=True,
):
    xname = "x"
    yname = "y"

    df = pd.DataFrame({xname: xplotdata, yname: yplotdata, "w": w})

    if not xrange[0] == None:
        df = df.drop(df[df[xname] < xrange[0]].index)
    if not xrange[1] == None:
        df = df.drop(df[df[xname] > xrange[1]].index)
    if not yrange[0] == None:
        df = df.drop(df[df[yname] < yrange[0]].index)
    if not yrange[1] == None:
        df = df.drop(df[df[yname] > yrange[1]].index)

    H, xedges, yedges = np.histogram2d(
        df[xname].to_numpy(),
        df[yname].to_numpy(),
        bins=(bins[0], bins[1]),
        weights=df["w"].to_numpy(),
    )
    X, Y = np.meshgrid(xedges, yedges)
    X, Y = np.meshgrid(xedges, yedges)
    H = H.T * const.e * 1e12

    if not ax == None:
        g = ax.pcolormesh(X, Y, H, cmap=cmap)
        if iscbar:
            cbar = plt.colorbar(g, ax=ax, pad=-0)
            ax.annotate("pC", xy=(1.09, 1), xycoords="axes fraction", color="black")

    return H, X, Y, xedges[:-1], yedges[:-1]



[docs]def laser_strength(I0: float, lambda0: float, normalised: bool = False) -> float:
    r"""Laser strength parameter :math:`a_0`.

    .. math::

        a_0 = \sqrt{\frac{e^2}{2 \pi^2 \epsilon_0 m_e^2 c^5} \lambda_0 I_0}

    or if :code:`normalised == True`:

    .. math::

        a_0 = 0.855 \lambda_0 [\mu \mathrm{m}] \sqrt{I_0 [10^{18} \mathrm{W/cm}^2]}

    Args:
        I0 (float): Maximum intensity
        lambda0 (float): Wavelength
        normalised (bool): Normalised equation. Default to False.

    Returns:
        float: Laser strength parameter
    """
    if normalised:
        return np.sqrt(
            (const.e**2)
            / (2 * np.pi**2 * const.epsilon_0 * const.m_e**2 * const.c**5)
            * lambda0
            * I0
        )
    else:
        return 0.855 * lambda0 * np.sqrt(I0)




[docs]def critical_density(lambda0: float) -> float:
    r"""Laser critical density :math:`n_{\mathrm{c}}`.

    .. math::

        n_{\mathrm{c}} = \frac{m_e \epsilon_0 \omega_0^2}{e^2}


    Args:
        lambda0 (float): Wavelength

    Returns:
        float: Laser critical density
    """
    return (
        const.m_e
        * const.epsilon_0
        / const.e**2
        * (2 * np.pi * const.c / lambda0) ** 2
    )




[docs]def accelerating_electric_field(ne: float) -> float:
    r"""Return the Rayleigh length :math:`z_{\rm R}`.

    .. math::

        E_{\rm max} = \frac{m_e c \omega_{\rm pe}}{e}

    Args:
        ne (float): Plasma density in m-3

    Returns:
        float: Maximum accelerating electric field
    """
    return const.m_e * const.c * plasma_frequency(ne) / const.e




[docs]def Rayleigh_length(w0: float, lambda0: float) -> float:
    r"""Return the Rayleigh length :math:`z_{\rm R}`.

    .. math::

        z_{\rm R} = \frac{\pi w_0^2}{\lambda_0}

    Args:
        w0 (float): Maximum waist
        lambda0 (float): Wavelength

    Returns:
        float: Rayleigh length
    """
    return np.pi * w0**2 / lambda0




[docs]def waist0_theory(z: float, w0: float, lambda0: float, zfoc: float = 0) -> float:
    r"""Return the theoretical waist over :math:`z`.

    .. math::

        w(z) = w_0 \sqrt{1 + \left( \frac{z - z_{\rm foc}}{z_{\rm R}} \right)^2}

    Args:
        z (array of float): Positions in z
        w0 (float): Maximum waist
        zfoc (float): Z focal position. Default to 0.
        lambda0 (float): Wavelength

    Returns:
        float: Theoretical waist
    """
    zr = Rayleigh_length(w0, lambda0)
    return w0 * np.sqrt(1.0 + ((z - zfoc) / zr) ** 2)




[docs]def plasma_frequency(ne: float) -> float:
    r"""Return the theoretical waist over :math:`z`.

    .. math::

        \omega_{\rm pe} = \sqrt{\frac{n_{\rm e} e^2}{m_{\rm e} \epsilon_0}}

    Args:
        ne (float): Plasma density in m-3

    Returns:
        float: Plasma frequency in rad/s.
    """
    return np.sqrt(ne * const.e**2 / const.m_e / const.epsilon_0)




[docs]def convert_laser_FWHM_duration(
    t: float, convert: str = "fbpic_to_FWHM_duration"
) -> float:
    """_summary_

    Args:
        t (float): time to convert.
        convert (str, optional): Type of conversion.
            Chose between ["fbpic_to_FWHM_duration", "FWHM_duration_to_fbpic", "smilei_to_FWHM_duration", "FWHM_duration_to_smilei"]. Defaults to "fbpic_to_FWHM_duration".

    Returns:
        float: Converted time
    """
    if convert == "fbpic_to_FWHM_duration":
        return t * np.sqrt(2 * np.log(2))

    elif convert == "FWHM_duration_to_fbpic":
        return t / np.sqrt(2 * np.log(2))

    if convert == "smilei_to_FWHM_duration":
        return t / np.sqrt(2)

    elif convert == "FWHM_duration_to_smilei":
        return t * np.sqrt(2)
